import { Code, Tabs, TabItem } from "@astrojs/starlight/components";

export const version = "0.0.2";
export const modules = [
  {
    name: "std",
    description: (
      <span>
        <code>std</code> is JSignal's standard component library. This is the
        recommended package if you are looking to start building GUIs quickly.
      </span>
    ),
  },
  {
    name: "ui",
    description: (
      <span>
        <code>ui</code> is JSignal's core interface library that provides
        windowing, event handling, rendering and layout. This is the recommended
        package if you are looking for a good foundation but want to implement
        all your own components.
      </span>
    ),
  },
  {
    name: "rx",
    description: (
      <span>
        <code>rx</code> is JSignal's core reactive library, providing reactive
        primitives and associated utilities.
      </span>
    ),
  },
];

JSignal offers three public packages with different levels of capabilities, each successive package is lower level than the previous:

<Tabs>
  {modules.map((module) => (
    <TabItem label={module.name}>
      {module.description}
      <br />
      <Tabs syncKey="buildSystem">
        <TabItem label="Maven">
          <Code
            code={`<dependency>
  <groupId>org.jsignal</groupId>
  <artifactId>${module.name}</artifactId>
  <version>${version}</version>
</dependency>"`}
            lang="xml"
          />
        </TabItem>
        <TabItem label="Gradle (Kotlin)">
          <Code
            code={`implementation("org.jsignal:${module.name}:${version}")`}
            lang="kotlin"
          />
        </TabItem>
        <TabItem label="Gradle (Groovy)">
          <Code
            code={`implementation 'org.jsignal:${module.name}:${version}'`}
            lang="groovy"
          />
        </TabItem>
      </Tabs>
    </TabItem>
  ))}
</Tabs>
