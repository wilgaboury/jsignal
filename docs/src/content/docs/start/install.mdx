JSignal offers three public packages with different capabilities, each one lower level than the previous:

<Tabs>
  <TabItem label="std">
    `std` is JSignal's standard component library. This is the reccomended package if you are looking to start building GUIs quickly.

    <Tabs syncKey="buildSystem">
      <TabItem label="Maven">
        ```xml
        <dependency>
          <groupId>org.jsignal</groupId>
          <artifactId>std</artifactId>
          <version>0.0.2</version>
        </dependency>
        ```
      </TabItem>
      <TabItem label="Gradle (Kotlin)">
        <Code code={'implementation("org.jsignal:std:0.0.2")'} lang="kotlin" />
      </TabItem>
      <TabItem label="Gradle (Groovy)">
        <Code code={"implementation 'org.jsignal:std:0.0.2'"} lang="groovy" />
      </TabItem>
    </Tabs>

  </TabItem>
  <TabItem label="ui">
    `ui` is JSignal's core interface library that provides windowing, events, rendering and layout. This is the reccomended package if you are looking for a good foundation but want to implement all your own widgets.

    <Tabs syncKey="buildSystem">
      <TabItem label="Maven">
        ```xml
        <dependency>
          <groupId>org.jsignal</groupId>
          <artifactId>ui</artifactId>
          <version>0.0.2</version>
        </dependency>
        ```
      </TabItem>
      <TabItem label="Gradle (Kotlin)">
        <Code code={'implementation("org.jsignal:ui:0.0.2")'} lang="kotlin" />
      </TabItem>
      <TabItem label="Gradle (Groovy)">
        <Code code={"implementation 'org.jsignal:ui:0.0.2'"} lang="groovy" />
      </TabItem>
    </Tabs>

  </TabItem>
  <TabItem label="rx">
    `rx` is JSignal's core reactive library, providing reactive primitives and associated utilities.
    <Tabs syncKey="buildSystem">
      <TabItem label="Maven">
        ```xml
        <dependency>
          <groupId>org.jsignal</groupId>
          <artifactId>rx</artifactId>
          <version>0.0.2</version>
        </dependency>
        ```
      </TabItem>
      <TabItem label="Gradle (Kotlin)">
        <Code code={'implementation("org.jsignal:rx:0.0.2")'} lang="kotlin" />
      </TabItem>
      <TabItem label="Gradle (Groovy)">
        <Code code={"implementation 'org.jsignal:rx:0.0.2'"} lang="groovy" />
      </TabItem>
    </Tabs>
  </TabItem>
</Tabs>
